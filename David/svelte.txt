<script>
	let x = 7;
	let num = 0
	
	function add() {
		num++
	}
	function sub() {
		num--
	}
</script>

<div id = "page">
	<h4>
		Press the button to increase
	</h4>
	<div id="buttons">
		<span><button on:click={add}>
				Increase
			</button> | <button on:click={sub}>
				Decrease
		</button></span>
	</div>
	<p>
		Number: {num}
	</p>
	{#if num > 10}
		<p>{num} is greater than 10</p>
	{:else if 5 > num}
		<p>{num} is less than 5</p>
	{:else}
		<p>{num} is between 5 and 10</p>
	{/if}
	{#if num % 2 == 0 && num != 0}
		<p>
			{num} is divisible by two
	</p>
	{/if}
	{#if num % 5 == 0 && num != 0}
		<p>
			{num} is divisible by five
	</p>
	{/if}
</div>
